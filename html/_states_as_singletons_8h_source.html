328 void changeMode(int modeNumber) {
329 std::cout << " RealTimeLoop::changeMode(" << modeNumber << ")\n";
330 switch (modeNumber) {
331 case 1:
332 currentMode = Mode1State::getInstance(this);
333 std::cout << " Transitioned to Mode1\n";
334 break;
335 case 2:
336 currentMode = Mode2State::getInstance(this);
337 std::cout << " Transitioned to Mode2\n";
338 break;
339 case 3:
340 currentMode = Mode3State::getInstance(this);
341 std::cout << " Transitioned to Mode3\n";
342 break;
343 default:
344 std::cout << " Invalid mode\n";
345 }
346 }
347
348 void eventX() {
349 if (currentMode) {
350 currentMode->eventX();
351 }
352 }
353
354 std::string getCurrentMode() const {
355 if (dynamic_cast<Mode1State*>(currentMode)) return "Mode1";
356 if (dynamic_cast<Mode2State*>(currentMode)) return "Mode2";
357 if (dynamic_cast<Mode3State*>(currentMode)) return "Mode3";
